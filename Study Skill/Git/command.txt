查看文件状态：
	$ git status

添加文件到暂存区：
	# 将“.”改为指定文件名可以只将某个文件添加到暂存区，而使用“.”是将所有文件添加到暂存区
	$ git add .

提交到仓库：
	$ git commit -m "此处是注释"

查看日志：
	$ git log
	# 扩展指令
	# --all 显示所有分支
	# --pretty=oneline 将提交信息显示为一行
	# --abbrev-commit 使得输出的commitld更简短
	# --graph 以图的形式显示

# 查看操作日志：
	$ git reflog

# 回退：
	# commitID是提交的唯一标识符,只要知道文件操作的id，都能到达那个版本，这个时候就需要找到那个id（可以使用git reflog查看操作日志）
	$ git reset --hard commitID

# 设置提交忽略列表：
	在当前目录下创建.gitignore文件，然后在里面写文件名称
	例如：*.a  # 意思是以“.a”结尾的文件都不提交

# 查看分支
	$ git branch

# 创建分支
	$ git branch 分支名

# 切换分支
	# 切换分支之后文件也会变，变成最后提交的那个版本，注意是提交的，没提交的不算
	# -b扩展命令可以创建并切换到分支
	$ git checkout 分支名

# 合并分支到分支
	# 合并时，需要切换到需要合并到的分支
	$ git merge 需要合并的分支名

# 删除分支
	# -d普通删除（需要检查），-D强制删除
	$ git branch -d 分支名
	$ git branch -D 分支名

# 生成公私秘钥
	$ ssh-keygen -t rsa

# 查看秘钥
	# cat 路径
	$ cat ~/.ssh/id_rsa.pub

# 验证秘钥是否成功
	# 我测试的是githup，连接超时，但是命令应该是这样的ssh -T git@托管网址
	$ ssh -T git@github.com
	# 22端口拒绝的话使用以下方法
	$ notepad ~/.ssh/config  # 创建config文件
	# 复制以下内容
-------------------------------------------
	Host github.com
    	Hostname ssh.github.com
    	Port 443
    	User git
   	IdentityFile ~/.ssh/id_rsa
-------------------------------------------
	# 保存重新尝试（目的更改端口为443）

# 添加远程仓库
	# origin是别名（请确保你在本地仓库打开的命令提示符）
	$ git remote add origin git@托管网址:仓库地址

# 查看当前所拥有的远程仓库
	$ git remote

# 推送到远程
	# 推送分支master到远程仓库origin的master分支上，如果名称一样可以直接省略冒号后面的；-f 强制覆盖
	$ git push origin master:master

# 移除废弃远程
	$ git remote remove 【name】

# 本地和远程绑定推送
	# 第一次推送要加--set-upstream添加关联关系
	$ git push --set-upstream origin master:master

# 查看本地与远程的绑定
	$ git branch -vv

# 克隆仓库
	$ git clone git@托管网址:仓库地址 自定义名称（可忽略）

# 抓取与拉取
	$ git fetch  # 将远程仓库的更新拉取到本地，但是不合并
	$ git pull  # 直接将远程仓库的更新更新到本地
